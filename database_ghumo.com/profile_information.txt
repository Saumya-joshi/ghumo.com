CREATE TABLE profile_information(profile_id int primary key AUTO_INCREMENT, title varchar(5) not null, F_name varchar(32) not null, M_name varchar(32), L_name varchar(32) not null, email_id varchar(50) not null, mobile_no varchar(10) not null, date_of_birth date not null, street_add varchar(75), pin_code varchar(6), city varchar(20), country varchar(32), state varchar(32), home_town varchar(32), marietal_status varchar(10), anniversary_date date, image_path varchar(200));
